// Firestore security rules for the 'chats' collection
// Place this in your Firebase console > Firestore Database > Rules
service cloud.firestore {
  match /databases/{database}/documents {
    // Chats collection (DMs)
    match /chats/{chatId} {
      allow read, write: if request.auth != null &&
        (request.auth.uid in resource.data.participants || request.auth.uid in request.resource.data.participants);
      
      // Messages subcollection
      match /messages/{messageId} {
        allow read, write: if request.auth != null &&
          (request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants);
      }
    }
    // Users collection (for user info)
    match /users/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    // Friends collection
    match /friends/{friendId} {
      allow read, write: if request.auth != null && request.auth.uid in resource.data.uids;
    }
    // Friend requests
    match /friendRequests/{requestId} {
      allow read, write: if request.auth != null && (request.auth.uid == resource.data.fromUid || request.auth.uid == resource.data.toUid);
    }
    // Community chats (groupes)
    match /communityChats/{groupId} {
      allow read, write: if request.auth != null;
      match /messages/{messageId} {
        allow read, write: if request.auth != null;
      }
      match /members/{memberId} {
        allow read, write: if request.auth != null;
      }
    }
  }
}

// Example structure for the 'chats' collection (DM):
// chats/{chatId}:
//   participants: [uid1, uid2]
//   participantsMeta: { uid1: {displayName, photoURL}, uid2: {displayName, photoURL} }
//   createdAt: Timestamp
//   lastMessageText: string (optional)
//   lastMessageAt: Timestamp (optional)
//
// chats/{chatId}/messages/{messageId}:
//   text: string
//   senderUid: string
//   createdAt: Timestamp
//   ... (other fields as needed)
